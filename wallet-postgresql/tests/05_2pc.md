# VALIDACIÓN DE 2PC

## Paso 1: Conectarse a shard1

docker exec -it pg_shard_1 psql -U admin -d walletdb

## Paso 2: Iniciar transacción

BEGIN;

UPDATE cuentas
SET saldo = saldo - 100000
WHERE id_cuenta = 1001;

PREPARE TRANSACTION 'tx_test';

---

## Paso 3: Conectarse a shard2

docker exec -it pg_shard_2 psql -U admin -d walletdb

BEGIN;

UPDATE cuentas
SET saldo = saldo + 100000
WHERE id_cuenta = 2001;

PREPARE TRANSACTION 'tx_test';

---

## Paso 4: Confirmar en ambos shards

COMMIT PREPARED 'tx_test';

---

## Paso 5: Validar saldos

SELECT * FROM cuentas WHERE id_cuenta = 1001;
SELECT * FROM cuentas WHERE id_cuenta = 2001;
